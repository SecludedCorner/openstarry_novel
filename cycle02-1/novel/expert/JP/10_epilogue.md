$$	ext{Design}(t_{	ext{now}}) \supsetneq 	ext{Implementation}(t_{	ext{now}})$$

設計は実装の真超集合である。差集合 $	ext{Design} \setminus 	ext{Implementation}$ は空ではない。ARCHIMEDES のロードマップとは、この差集合を空集合へと縮小させる計画なのだ。

 * ISensation — 受蘊ルートインターフェース (Vedanā-skandha Root Interface)
 *
 * @ruling D1 — VedanaAssessment 二層構造 (sensor + advisory controller)
 * @ruling D2 — Tick同期 PID + イベントレベルの vedana タグ

 *
 * @see aggregates.ts ルートインターフェースの定義
 * @see debates_and_synthesis.md 論争の完全な記録
 */
export interface ISensation extends IOpenStarryPlugin {
  readonly skandha: 'vedana';

  /**
   * 完全な3チャンネル PID 評価を実行する。
   * ExecutionLoop の各ティックにつき一度だけ呼び出される (Tick同期)。
   *
   * @ruling D1 — センサー出力 (受動的) とコントローラ助言 (助言的、非拘束) の両方を返す
   * @ruling D2 — これがオーソリティとなる vedana の読み取り値である
   *
   * @returns 二層構造を持つ VedanaAssessment:
   *   レイヤー 1: センサー (苦/楽/捨の数値 + 信号)
   *   レイヤー 2: コントローラ (VedanaRecommendation — 助言的)
   */

  /**
   * PID 計算のための生システムメトリクスを取り込む。
   *
   * @param metrics — MetricsCollector からのキー・バリューペア
   */
  ingestMetrics(metrics: Record<string, number>): void;

  /**
   * vedana 評価のためにツールの実行結果を取り込む。
   *
   * @ruling D1 — 純粋なセンシングであり、副作用はない
   */
  ingestToolResult(
    toolName: string,
    isError: boolean,
    durationMs: number
  ): void;

  /**
   * vedana 評価のために LLM レスポンスのメタデータを取り込む。
   */
  ingestLLMResult(
    tokenCount: number,
    stopReason: string
  ): void;

  /**
   * 任意の vedana チャンネルにおける閾値越えを購読する。
   *
   * @ruling D2 — 閾値ベースの通知メカニズム

   * @param threshold — トリガー閾値 (0.0-1.0)
   * @param callback — 閾値を越えた際に呼び出される
   * @returns 購読解除関数
   */
  onThreshold(
    channel: VedanaType,
    threshold: number,
    callback: (signal: VedanaSignal) => void
  ): () => void;

  /**
   * イベントタギングのための軽量な vedana タグを取得する。
   * O(1) キャッシュクエリ — 直近の assess() 結果から導出。
   *
   * @ruling D2 — すべてのイベントは vedana タグを保持する (遍行の原則)
   * @ruling D2 — タグは導出されるものであり、イベントごとに再計算はされない
   */
  getVedanaTag(): VedanaTag;
}

 * @ruling D1 — センサーとコントローラの概念的分離
 */
interface VedanaAssessment {
  // ═══ レイヤー 1: センサー出力 (純粋な観察) ═══
  readonly dukkha: number;      // 0.0-1.0 — 苦受 (suffering)
  readonly sukha: number;       // 0.0-1.0 — 楽受 (joy)
  readonly upekkha: number;     // 0.0-1.0 — 捨受 (equanimity)
  readonly signals: readonly VedanaSignal[];
  readonly timestamp: number;

  // ═══ レイヤー 2: コントローラの助言 (助言的) ═══
  readonly controlOutput: number;
  readonly recommendation: VedanaRecommendation;
  readonly recommendationFreshness:
    | 'current'    // このティックのデータから計算
    | 'cached'     // 前のティックから引き継ぎ
    | 'default';   // 初期状態 (upekkha)
}
